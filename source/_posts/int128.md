## 特点
1. 不支持printf，scanf，cin，cout，自己输入输出
2. 至多可存38位
3. 运算同int

<!--more-->

```cpp
#include<iostream>
using namespace std;
__int128 read()
{
  __int128 x=0,f=1;
    char ch=getchar();
    while(ch<'0'||ch>'9'){
        if(ch=='-')
            f=-1;
        ch=getchar();
    }
    while(ch>='0'&&ch<='9'){
        x=x*10+ch-'0';
        ch=getchar();
    }
    return x*f;
}
__int128 print(_int128 x)
{
  if(x < 0)
    {
      putchar(' - ');
      x =- x;
    }
  if(x > 9)
    {
      print(x / 10);
    }
  putchar(x % 10 + '0');
}
int main()
{
  __int128 a, b;
  a = read();
  b = read();
  print(a + b);
  cout << endl;
}
```
